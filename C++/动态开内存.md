# 动态开内存

## code:
```cpp
#include<cstdio>
#include<iostream>
using namespace std;
void print(int *x)
{
	printf("%d %d %d %d %d \n",*x,*x+1,*x+2,*x+3,*x+4);
}
int main()
{
	int *a=new int[5];
	printf("\nint *a=new int[5];\n");
	print(a);
	
	int *b=new int[5]{};
	printf("\nint *b=new int[5]{};\n");
	print(b);
	
	int *c=new int[5]();
	printf("\nint *c=new int[5]();\n");
	print(c);
	
	int *d=new int[5]{5};
	printf("\nint *d=new int[5]{5};\n");
	print(d);
	
	delete[] a;
	printf("\ndelete[] a;\n");
	print(a);
	
	printf("\n");
	printf("\nchar *s=\"Minecraft Is Love, Minecraft Is Life!\";\n");
	char *s="Minecraft Is Love, Minecraft Is Life!";
	cout<<s<<endl;
}
```
