# 树链剖分


[P3384 【模板】重链剖分/树链剖分](https://www.luogu.com.cn/problem/P3384)

```cpp
int fa[N],son[N],sz[N],dfn[N],cnt,top[N],nfd[N],dep[N];
void dfs1(int x,int ft)
{
	fa[x]=ft;
	sz[x]=1;
	dep[x]=dep[ft]+1;
	for(auto to:v[x])
	{
		if(to==ft) continue;
		dfs1(to,x);
		sz[x]+=sz[to];
		son[x]=sz[to]>sz[son[x]]?to:son[x];
	}
}
void dfs2(int x,int tp)
{
	dfn[x]=++cnt;
	nfd[cnt]=x;
	top[x]=tp;
	if(son[x]) dfs2(son[x],tp);
	for(auto to:v[x])
	{
		if(to==fa[x]||to==son[x]) continue;
		dfs2(to,to);
	}
}

int chain_find(int x,int y)
{
	int ans=0;
	while(top[x]!=top[y])
	{
		if(dep[top[x]]<dep[top[y]]) swap(x,y);
		ans+=find(1,1,n,dfn[top[x]],dfn[x]);
		x=fa[top[x]];
	}
	if(dep[x]>dep[y]) swap(x,y);
	ans+=find(1,1,n,dfn[x],dfn[y]);
	return ans;
}
```
