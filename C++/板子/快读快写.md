# 快读/快写


```cpp
int read()
{
	char rc=getchar();
	int rf=1,rx=0;
	while(!isdigit(rc))
	{
		if(rc=='-') rf=-1;
		rc=getchar();
	}
	while(isdigit(rc))
	{
		rx=rx*10+rc-'0';
		rc=getchar();
	}
	return rx*rf;
}
void write(int x)
{
	if(x<0)
	{
		putchar('-');
		write(-x);
		return;
	}
	if(x/10) write(x/10);
	putchar(x%10+'0');
}
```
